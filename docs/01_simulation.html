# greedy algorithm to discard redundant SNPs
prune_ld <- function(R, thresh) {
  keep <- rep(TRUE, nrow(R))
  for (i in seq_len(nrow(R) - 1)) {
    if (!keep[i]) next
    for (j in (i + 1):nrow(R)) {
      if (keep[j] && R[i, j]^2 > thresh) keep[j] <- FALSE
    }
  }
  keep
}

# runs linear regression for every SNP to measure correlation with phenotype y
get_marginal_stats <- function(X, y) {
  # Vectorized OLS for massive speedup
  m <- ncol(X)
  yc <- y - mean(y)
  Xc <- sweep(X, 2, colMeans(X), "-")
  sxx <- colSums(Xc^2)
  sxy <- as.numeric(crossprod(Xc, yc))
  beta <- sxy / sxx
  syy <- sum(yc^2)
  rss <- pmax(syy - beta * sxy, 0)
  se <- sqrt((rss / (length(y) - 2)) / sxx)
  z <- beta / se
  z[!is.finite(z)] <- 0
  list(beta = beta, z = z)
}
# t_burden from problem statement
t_burden <- function(w, Z, R) {
  denom <- sqrt(as.numeric(t(w) %*% R %*% w))
  if (!is.finite(denom) || denom <= 0) return(0)
  as.numeric((t(w) %*% Z) / denom)
}
# scans across chromosome, testing windows at every pivotal SNIP and ouptput RAS
screen_forward_max_region_ss <- function(b_disc, z_targ, R, mask) {
  N <- length(b_disc)
  this.seq <- seq(1, N, by = skip1)
  sub.seq  <- c(0, seq(min_window_size, max_window_size, by = skip2))
  if (sub.seq[length(sub.seq)] != max_window_size) sub.seq <- c(sub.seq, max_window_size)
  y_profile <- sapply(this.seq, function(site) {
    best_p <- 1
    for (ws in sub.seq) {
      idx <- if (ws == 0) site else max(1, site - ws):min(N, site + ws)
      idx <- idx[mask[idx]]
      if (length(idx) < 1) next
      w_sub <- b_disc[idx]
      z_sub <- z_targ[idx]
      R_sub <- R[idx, idx, drop = FALSE]
      num <- sum(w_sub * z_sub)
      denom <- sqrt(as.numeric(w_sub %*% R_sub %*% w_sub))
      if (!is.finite(denom) || denom <= 0) next
      
      t_val <- num / denom
      best_p <- min(best_p, 2 * pnorm(-abs(t_val)))
    }
    -log10(max(best_p, .Machine$double.xmin))
  })
  list(x = this.seq, y = y_profile)
}

# splits. No summary-stat analog exists (Method A has fixed discovery/
# target summary stats, nothing to resplit). This runs single-pass instead
run_one_replicate <- function(true_beta, seed) {
  set.seed(seed)
  X_d <- sim_genotypes(n_disc, R_true)
  X_t <- sim_genotypes(n_targ, R_true)
  y_d <- X_d %*% true_beta + rnorm(n_disc, 0, 3)
  y_t <- X_t %*% true_beta + rnorm(n_targ, 0, 3)
  disc <- get_marginal_stats(X_d, y_d)
  targ <- get_marginal_stats(X_t, y_t)
  scan <- screen_forward_max_region_ss(disc$beta, targ$z, R_empirical, pruned_mask)
  list(scan = scan, X_t = X_t, y_t = y_t, b_disc = disc$beta)
}

# scw = 8 fixes this, as acceptance goes to ~100%
detect <- function(scan, window_size, scw = 8) {
  stopifnot(window_size < length(scan$x)) # make sure window size is less than chromosome size
  out <- tryCatch({
    # run changepoint detection algorithm (from old package)
    cp <- ras_detect(
      x = scan$x, y = scan$y,
      window_size                    = window_size,
      slope_check_window_size        = scw,
      slope.p.values.threshold.left  = slope_thresh,
      slope.p.values.threshold.right = slope_thresh
    )
    # run Davies test to prove change points are statistically significant (also from old package)
    val <- ras_validate(
      this.result = cp, x = scan$x, y = scan$y,
      this.start = 1, this.skip = skip1,
      second_window_size = 15, min_signal = 2.5,
      p.value.threshold = davies_thresh
    )
    # returns locations of validated peaks (tau_hats)
    list(val = val, raw_candidates = cp$all.changepoints, error = FALSE)
  }, error = function(e) {
    list(val = list(tau_hats = numeric(0)), raw_candidates = NULL, error = TRUE)
  })
  out
}

  ws <- window_candidates[k]

  null_res <- future_lapply(seq_len(n_sweep), function(s) {
    # this is null hypothesis, so any gene found here is Type I error

    scan <- run_one_replicate(signal_beta, seed = 20000 * ws + s)$scan

      center = any(tau >= 130 & tau <= 170),

  sweep_results$null_errors[k]     <- null_err
  sweep_results$power_errors[k]    <- power_err
