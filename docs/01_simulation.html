prune_ld <- function(R, thresh) {
  keep <- rep(TRUE, nrow(R))
  for (i in seq_len(nrow(R) - 1)) {
    if (!keep[i]) next
    for (j in (i + 1):nrow(R)) {
      if (keep[j] && R[i, j]^2 > thresh) keep[j] <- FALSE
    }
  }
  keep
}

get_marginal_stats <- function(X, y) {
  m <- ncol(X)
  beta <- se <- numeric(m)
  yc <- y - mean(y)
  for (j in seq_len(m)) {
    xj  <- X[, j] - mean(X[, j])
    sxx <- sum(xj^2)
    b   <- sum(xj * yc) / sxx
    rss <- max(0, sum(yc^2) - b * sum(xj * yc))
    se[j]   <- sqrt((rss / (length(y) - 2)) / sxx)
    beta[j] <- b
  }
  list(beta = beta, z = beta / se)
}

t_burden <- function(w, Z, R, lam = ridge_lambda) {
  Rl    <- R + lam * diag(nrow(R))
  denom <- sqrt(as.numeric(t(w) %*% Rl %*% w))
  if (!is.finite(denom) || denom <= 0) return(0)
  as.numeric((t(w) %*% Z) / denom)
}

screen_forward_max_region_ss <- function(b_disc, z_targ, R, mask) {
  N <- length(b_disc)
  this.seq <- seq(1, N, by = skip1)
  sub.seq  <- c(0, seq(min_window_size, max_window_size, by = skip2))
  if (sub.seq[length(sub.seq)] != max_window_size) sub.seq <- c(sub.seq, max_window_size)

  y_profile <- numeric(length(this.seq))
  for (i in seq_along(this.seq)) {
    site   <- this.seq[i]
    best_p <- 1
    for (ws in sub.seq) {
      idx <- if (ws == 0) site else max(1, site - ws):min(N, site + ws)
      idx <- idx[mask[idx]]
      if (length(idx) < 1) next
      t_val  <- t_burden(b_disc[idx], z_targ[idx], R[idx, idx, drop = FALSE])
      best_p <- min(best_p, 2 * pnorm(-abs(t_val)))
    }
    y_profile[i] <- -log10(max(best_p, .Machine$double.xmin))
  }
  list(x = this.seq, y = y_profile)
}

# splits. No clean summary-stat analog exists (Method A has fixed discovery/
# target summary stats, nothing to resplit) -- flagged as an open design
# question in 00_problem_statement.Rmd. This runs single-pass.
run_one_replicate <- function(true_beta, seed) {
  set.seed(seed)
  X_d <- sim_genotypes(n_disc, R_true)
  X_t <- sim_genotypes(n_targ, R_true)
  y_d <- X_d %*% true_beta + rnorm(n_disc, 0, 3)
  y_t <- X_t %*% true_beta + rnorm(n_targ, 0, 3)
  disc <- get_marginal_stats(X_d, y_d)
  targ <- get_marginal_stats(X_t, y_t)
  scan <- screen_forward_max_region_ss(disc$beta, targ$z, R_empirical, pruned_mask)
  list(scan = scan, X_t = X_t, y_t = y_t, b_disc = disc$beta)
}

# *** CRITICAL PARAMETER: scw defaults to 8, NOT 5. ***
# slope_check_window_size = 5 (the package's typical default) was found to
# give only ~17-20% first-pass acceptance on true signal, because the local
# slope re-check window sits inside the flat top of the causal plateau.
# scw = 8 fixes this: acceptance goes to ~100%. VERIFY this default is 8
# before running anything long -- this is the exact parameter that was
# accidentally lost once already.
detect <- function(scan, window_size, scw = 8) {
  stopifnot(window_size < length(scan$x))
  out <- tryCatch({
    cp <- ras_detect(
      x = scan$x, y = scan$y,
      window_size                    = window_size,
      slope_check_window_size        = scw,
      slope.p.values.threshold.left  = slope_thresh,
      slope.p.values.threshold.right = slope_thresh
    )
    val <- ras_validate(
      this.result = cp, x = scan$x, y = scan$y,
      this.start = 1, this.skip = skip1,
      second_window_size = 15, min_signal = 2.5,
      p.value.threshold = davies_thresh
    )
    list(val = val, raw_candidates = cp$all.changepoints, error = FALSE)
  }, error = function(e) {
    list(val = list(tau_hats = numeric(0)), raw_candidates = NULL, error = TRUE)
  })
  out
}

indiv_baseline_profile <- function(scan, X_t, y_t, b_disc) {
  sub.seq <- c(0, seq(min_window_size, max_window_size, by = skip2))
  if (sub.seq[length(sub.seq)] != max_window_size) sub.seq <- c(sub.seq, max_window_size)
  sapply(scan$x, function(site) {
    best_p <- 1
    for (ws in sub.seq) {
      idx <- if (ws == 0) site else max(1, site - ws):min(n_snps, site + ws)
      idx <- idx[pruned_mask[idx]]
      if (length(idx) < 1) next
      LPRS <- X_t[, idx, drop = FALSE] %*% b_disc[idx]
      fit  <- summary(lm(y_t ~ LPRS))
      p    <- if (nrow(fit$coefficients) > 1) fit$coefficients[2, 4] else 1
      best_p <- min(best_p, p)
    }
    -log10(max(best_p, .Machine$double.xmin))
